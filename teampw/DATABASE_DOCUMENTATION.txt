# Пользователь
User:
	name (уникальное имя аккаунта)
	email (почта от аккаунта)
	password (пароль от аккаунта в хэше md5)

# Команда
Team:
	name (уникальное имя команды)
	info (инфомрация о команде)
	owner (внешний ключ. Объект владельца)

# Неподтвержденный аккаунт
UnverifiedUser:
	name (уникальное имя аккаунта)
	email (почта аккаунта)
	password (пароль от аккаунта в хэше md5)
	code (код для подтверждения почты)

# Принадлежность пользователя к команде
TeamMembership:
	user (внешний ключ. Объект пользователя)
	team (внешний ключ. Объект команды)

# Анкета команды
TeamAd:
	team (внешний ключ. Объект команды)
	project (направление, в котором команда хочет развиваться)
	role (кого ищет команда)
	skills (какие навыки должны быть у будущих участников)

# Анкета пользователя
UserAd:
	user (внешний ключ. Объект пользователя)
	project (направление, в котором пользователь хочет развиваться)
	role (кем хочет быть пользователь)
	skills (навыки пользователя)

# Приглашение в команду
Invite:
	sender (внешний ключ. Объект пользователя что отправил приглашение)
	receiver (внешний ключ. Объект пользователя кому адресовано приглашение)
	team (внешний ключ. Объект команды, к которой получатель присоединится)

# Запрос на вступление в команду
Request:
	sender (внешний ключ. Объект пользователя что отправил запрос)
	receiver (внешний ключ. Объект пользователя кому адресован запрос)
	team (внешний ключ. Объект команды, к которой отправитель хочет присоединится)
